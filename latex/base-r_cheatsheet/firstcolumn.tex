\begin{block}{Getting Help}
  
  \vskip+0.5ex
  \blocksubtitle{Accessing the help files}
  \vskip+0.5ex
  \inline{?mean}\\{\small Get help of a particular function.}\br
  \inline{help.search('weighted mean')}\\{\small Search the help files for a word or phrase.}\br
  \inline{help(package = 'dplyr')}\\{\small Find help for a package.}
  \vskip+2ex

  \blocksubtitle{More about an object}
  \vskip+0.5ex
  \inline{str(iris)}\\{\small Get a summary of an object's structure.}\br
  \inline{class(iris)}\\{\small Find the class an object belongs to.}
\end{block}

\begin{block}{Using Packages}
  \vskip+0.5ex
  \inline{install.packages('dplyr')}\\{\small Download and install a package from CRAN.}\br
  \inline{library(dplyr)}\\{\small Load the package into the session, making all its functions available to use.}\br
  \inline{dplyr::select}\\{\small Use a particular function from a package.}\br
  \inline{data(iris)}
\end{block}

\begin{block}{Working Directory}
  \vskip+0.5ex
  \inline{getwd()}\\{\small Find the current working directory (where inputs are found and outputs are sent).}\br
  \inline{setwd('C://file/path')}\\{\small Change the current working directory.}\br
  \vskip+2ex
  \textbf{Use projects in RStudio to set the working directory to the folder you are working in.}
\end{block}
